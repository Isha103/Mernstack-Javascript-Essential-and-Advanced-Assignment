function checkNumber(num) { if (num > 0) { console.log(num + " is a positive number."); } else if (num < 0) { console.log(num + " is a negative number."); } else { console.log("The number is zero."); } } // Example usage: checkNumber(5); // Output: 5 is a positive number. checkNumber(-3); // Output: -3 is a negative number. checkNumber(0); // Output: The number is zero.
